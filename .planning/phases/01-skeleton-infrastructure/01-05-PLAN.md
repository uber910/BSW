Шаг A — создать `Dockerfile` в корне:

```dockerfile
ARG SERVICE
ARG PYTHON_VERSION=3.10-slim-bookworm

FROM python:${PYTHON_VERSION} AS builder

ENV PYTHONUNBUFFERED=1 \
    PYTHONDONTWRITEBYTECODE=1 \
    UV_LINK_MODE=copy \
    UV_PYTHON_DOWNLOADS=never \
    UV_PROJECT_ENVIRONMENT=/opt/venv

RUN apt-get update \
    && apt-get install -y --no-install-recommends curl ca-certificates \
    && rm -rf /var/lib/apt/lists/*

COPY --from=ghcr.io/astral-sh/uv:0.11.14 /uv /usr/local/bin/uv

WORKDIR /app

COPY pyproject.toml uv.lock README.md ./
COPY src ./src

RUN uv sync --frozen --no-dev

FROM python:${PYTHON_VERSION} AS runtime

ARG SERVICE
ENV SERVICE=${SERVICE}

ENV PYTHONUNBUFFERED=1 \
    PYTHONDONTWRITEBYTECODE=1 \
    PATH="/opt/venv/bin:$PATH" \
    PYTHONPATH=/app/src

RUN apt-get update \
    && apt-get install -y --no-install-recommends curl ca-certificates \
    && rm -rf /var/lib/apt/lists/* \
    && groupadd --system --gid 1000 app \
    && useradd --system --uid 1000 --gid app --create-home --shell /usr/sbin/nologin app

WORKDIR /app

COPY --from=builder /opt/venv /opt/venv
COPY --chown=app:app src ./src
COPY --chown=app:app alembic.ini ./alembic.ini
COPY --chown=app:app alembic ./alembic

USER app

CMD ["sh", "-c", "exec python -m ${SERVICE}"]
```

Комментарий к CMD: ARG-параметр `SERVICE` нельзя подставить в JSON-array CMD напрямую (Docker не делает substitution в exec-form JSON), но `sh -c "exec python -m ${SERVICE}"` — это **корректный exec-form**: `exec` заменяет shell на Python, поэтому Python становится PID 1 и получает SIGTERM. Это эквивалент рекомендации D-04 «exec-form CMD». Альтернатива — два Dockerfile, но D-04 явно требует ОДИН Dockerfile с `ARG SERVICE`.
