    ```python
        D-16 (Pitfall A3 mitigation): pool_size=10, max_overflow=20 — enough

        pool_pre_ping=True — issues SELECT 1 on checkout, kills stale connections

            pool_size=10,
            max_overflow=20,
            pool_pre_ping=True,
            pool_recycle=1800,
        )
        sessionmaker = async_sessionmaker(engine, expire_on_commit=False)
        return engine, sessionmaker
    ```

    ```python
    from __future__ import annotations

    import logging

    import structlog
    from sqlalchemy import text
    from sqlalchemy.exc import SQLAlchemyError
    from sqlalchemy.ext.asyncio import AsyncEngine
    from tenacity import (
        before_sleep_log,
        retry,
        stop_after_attempt,
        wait_exponential,
    )

    log = structlog.get_logger()
    _stdlib_log = logging.getLogger(__name__)


    @retry(
        stop=stop_after_attempt(10),
        wait=wait_exponential(multiplier=1, min=1, max=10),
        reraise=True,
        before_sleep=before_sleep_log(_stdlib_log, logging.WARNING),
    )
    async def wait_for_postgres(engine: AsyncEngine) -> None:
        """Block until PG accepts SELECT 1, or give up after 10 attempts.

        D-27: tenacity @retry automatically uses AsyncRetrying for coroutines.
        Cumulative wait ≈ 1+2+4+8+10+10+10+10+10+10 = ~75s worst case before
        reraise. Lifespan (Plan 03-08) calls this — if all 10 attempts fail,
        startup crashes with a clear `OperationalError` and structlog log.

        Pitfall D2 mitigation: surfaces bad DSN at startup, not at first request.
        """
        async with engine.connect() as conn:
            await conn.scalar(text("SELECT 1"))


    async def ping_postgres(engine: AsyncEngine) -> bool:
        """Single SELECT 1 ping for /health — no retry, returns True/False.

        D-26: per-request check (no caching at test-task scale). ~5ms overhead
        is fine; live 503 is critical for docker-compose healthcheck.
        D-29: SQLAlchemyError caught → return False → route emits 503.
        Bare Exception is NOT caught — non-SQLAlchemy errors should propagate
        and cause 500 (info disclosure risk to swallow them silently).
        """
        try:
            async with engine.connect() as conn:
                await conn.scalar(text("SELECT 1"))
            return True
        except SQLAlchemyError as exc:
            log.warning("health.check.failed", check="postgres", error=str(exc))
            return False
    ```

    Ключевые детали:
    - `before_sleep_log(_stdlib_log, logging.WARNING)` — tenacity hook логирует через stdlib logging (структурное wrapping в structlog handled на root logger level в config).
    - Bare `Exception` НЕ ловим — только `SQLAlchemyError`. Это сознательное решение: если упало что-то совсем другое (например asyncio.CancelledError), пусть пробрасывается.
    - `log.warning("health.check.failed", check="postgres", error=str(exc))` логирует ТОЛЬКО `str(exc)` — не сам engine repr (DSN credentials leak risk).

    ```python
    BM-02 / D-15 / D-16: create_engine_and_sessionmaker returns engine + sessionmaker
    with locked params (pool_size=10, max_overflow=20, pool_pre_ping=True,

    ping returns False after engine.dispose() simulates PG outage.
    """

    from __future__ import annotations

    from sqlalchemy.ext.asyncio import (
        AsyncEngine,
        async_sessionmaker,
    )

    from bet_maker.infrastructure.db.engine import create_engine_and_sessionmaker
    from bet_maker.infrastructure.db.pings import ping_postgres
    from bet_maker.settings.config import BetMakerSettings


    class TestCreateEngineAndSessionmaker:
        """D-15 + D-16: engine + sessionmaker built with locked params."""

        def test_returns_engine_and_sessionmaker_tuple(self, pg_dsn: str) -> None:
            """Factory returns 2-tuple (AsyncEngine, async_sessionmaker)."""
            settings = BetMakerSettings(postgres_dsn=pg_dsn)  # type: ignore[call-arg]
            engine, maker = create_engine_and_sessionmaker(settings)
            try:
                assert isinstance(engine, AsyncEngine)
                assert isinstance(maker, async_sessionmaker)
            finally:
                # cleanup — synchronous close not needed for test (engine disposed at session end)
                pass

        async def test_sessionmaker_has_expire_on_commit_false(
            self, pg_dsn: str
        ) -> None:
            """D-15: sessionmaker MUST be expire_on_commit=False (A1 mitigation)."""
            settings = BetMakerSettings(postgres_dsn=pg_dsn)  # type: ignore[call-arg]
            engine, maker = create_engine_and_sessionmaker(settings)
            try:
                # introspect kwargs captured by async_sessionmaker
                # SQLAlchemy 2.0.x exposes via `_session_class.expire_on_commit` or kw dict
                async with maker() as session:
                    assert session.sync_session.expire_on_commit is False
            finally:
                await engine.dispose()

        async def test_engine_pool_params_locked(self, pg_dsn: str) -> None:
            """D-16: pool_size=10, max_overflow=20, pool_pre_ping=True, pool_recycle=1800."""
            settings = BetMakerSettings(postgres_dsn=pg_dsn)  # type: ignore[call-arg]
            engine, _ = create_engine_and_sessionmaker(settings)
            try:
                pool = engine.pool
                # QueuePool exposes .size() and ._max_overflow / ._recycle in private API
                assert pool.size() == 10, f"expected pool_size=10, got {pool.size()}"
                # _max_overflow and _recycle are stable across SQLA 2.0.x
                assert pool._max_overflow == 20, pool._max_overflow  # type: ignore[attr-defined]
                assert pool._recycle == 1800, pool._recycle  # type: ignore[attr-defined]
                # pool_pre_ping is on the dialect
                assert engine.pool._pre_ping is True  # type: ignore[attr-defined]
            finally:
                await engine.dispose()


    class TestPingPostgres:
        """BM-08 / D-26 / D-29: ping_postgres contract."""

        async def test_ping_returns_true_on_healthy_engine(
            self, async_engine: AsyncEngine
        ) -> None:
            """Healthy PG (testcontainers) — SELECT 1 succeeds → True."""
            result = await ping_postgres(async_engine)
            assert result is True

        async def test_ping_returns_false_on_disposed_engine(
            self, pg_dsn: str
        ) -> None:
            """D-29: disposed engine → SQLAlchemyError → False (NOT raise).

            Build a throwaway engine, dispose it, then ping. The disposed engine's
            pool cannot establish new connections — SQLAlchemyError is raised
            internally, ping_postgres catches it and returns False.
            """
            settings = BetMakerSettings(postgres_dsn=pg_dsn)  # type: ignore[call-arg]
            engine, _ = create_engine_and_sessionmaker(settings)
            # Force the pool to be unusable: dispose immediately, then try to ping
            # against a bad DSN by replacing pool — simpler: use a non-routable DSN
            await engine.dispose()

            finally:
    ```
